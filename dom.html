JavaScript Manipulation of CSS Properties

Changing the background color

function changeColor() {
	// Get the color input by the user and format it as a hexadecimal number
	var color = document.getElementById("input").value;
	var hexcolor = "#" + color;

	// Apply the color to the CSS attribute of the div
	var div = document.getElementById("div1");
	div.style.backgroundColor = hexcolor;
}
