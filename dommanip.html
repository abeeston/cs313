function addToList() {

	// Get the value of the input and get the list so we don't have to keep writing these
	var input = document.getElementById("input").value;
	var list = document.getElementById("mythings");

	// Create a new element (a list item)
	var item = document.createElement("li");

	// Create a new text node
	var textnode = document.createTextNode(input);

	// Append the text node as a child of the list item (as per DOM structure)
	item.appendChild(textnode);

	// Insert the list item into the list before the 0th child node (in this case it starts out empty)
	list.insertBefore(item, list.childNodes[0]);

	// Clear the input and refocus to the input field
	document.getElementById("input").value = '';
	document.getElementById("input").focus();
}

function remChild() {
	var list = document.getElementById("mythings");

	// It is good practice to test if children nodes exist before trying to delete any
	if (list.hasChildNodes()) {
		// remove the child node at the top of the list
		list.removeChild(list.childNodes[0]);
	} else {
		alert("There are no items to delete");
	}
}

function remAllChildren() {
	var list = document.getElementById("mythings");

	// While children nodes are still present
	while (list.hasChildNodes()) {
		// remove the child node at the top of the list
		list.removeChild(list.childNodes[0]);
	}
}
